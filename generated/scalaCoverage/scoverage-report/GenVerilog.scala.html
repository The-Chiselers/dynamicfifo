27     val thisExternalRam = paramVec(0).asInstanceOf[Boolean]
28     val thisDataWidth   = paramVec(1).asInstanceOf[Int]
29     val thisFifoDepth   = paramVec(2).asInstanceOf[Int]

40         s"fifoDepth = $thisFifoDepth"
41     )
42 
43     // Generate basic Verilog (suppress SV features with lowering, etc)
44     ChiselStage.emitSystemVerilog(
45       new DynamicFifo(myParams),
46       firtoolOpts = Array(
47         "--lowering-options=disallowLocalVariables,disallowPackedArrays",
48         "--disable-all-randomization",
49         "--strip-debug-info",
