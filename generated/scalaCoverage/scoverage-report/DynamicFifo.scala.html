14 /** A synchronous FIFO and FIFO controller with dynamic flags
15   *
16   * @constructor
17   *   create a new FIFO or FIFO controller
18   * @param externalRam
19   *   creates just the FIFO controller (uses ext. SRAM)
20   * @param dataWidth
21   *   defines the data width of the FIFO
22   * @param fifoDepth
23   *   defines the depth of the FIFO (must be a power of 2)
24   * @author
25   *   Warren Savage
26   * @todo
27   *   emit license from FIRRTL
28   * @see
29   *   [[http://www.rocksavage.tech/]] for more information
30   *
31   * <img src="doc/images/user-guide/block-diagram.png" />
32   */
33 class DynamicFifo(p: BaseParams) extends Module {

35   val io = IO(new Bundle {
36     val push             = Input(Bool())
37     val pop              = Input(Bool())
38     val dataIn           = Input(UInt(p.dataWidth.W))
39     val dataOut          = Output(UInt(p.dataWidth.W))
40     val empty            = Output(Bool())
41     val full             = Output(Bool())
42     val almostEmpty      = Output(Bool())
43     val almostFull       = Output(Bool())
44     val almostEmptyLevel = Input(UInt(log2Ceil(p.fifoDepth).W))
45     val almostFullLevel  = Input(UInt(log2Ceil(p.fifoDepth).W))
46     val ramWriteEnable   = Output(Bool())
47     val ramWriteAddress  = Output(UInt(log2Ceil(p.fifoDepth).W))
48     val ramDataIn        = Output(UInt(p.dataWidth.W))
49     val ramReadEnable    = Output(Bool())
50     val ramReadAddress   = Output(UInt(log2Ceil(p.fifoDepth).W))
51     val ramDataOut       = Input(UInt(p.dataWidth.W))
52   })
53 
54   val fifoMemory = Reg(Vec(p.fifoDepth, UInt(p.dataWidth.W)))
55 
56   val tail  = RegInit(0.U(log2Ceil(p.fifoDepth + 1).W))
57   val head  = RegInit(0.U(log2Ceil(p.fifoDepth + 1).W))
58   val count = RegInit(0.U(log2Ceil(p.fifoDepth + 1).W))
59 
60   when(io.push === 1.U) { count := count + 1.U }
61   when(io.pop === 1.U) { count := count - 1.U }
62   when((io.pop === 1.U) && io.push === 1.U) { count := count }
63 
64   io.empty       := count === 0.U
65   io.full        := count === p.fifoDepth.U
66   io.almostEmpty := count <= io.almostEmptyLevel
67   io.almostFull  := count >= io.almostFullLevel
68 
69   when(io.pop && !io.empty) {
70     tail := (tail + 1.U)
71   }
72 
73   when(io.push && !io.full) {
74     fifoMemory(head) := io.dataIn
75     head             := (head + 1.U)
76   }
77 
78   io.dataOut         := Mux(p.externalRam.B, io.ramDataOut, fifoMemory(tail))
79   io.ramDataIn       := io.dataIn
80   io.ramReadAddress  := tail
81   io.ramWriteAddress := head
82   io.ramReadEnable   := io.pop
83   io.ramWriteEnable  := io.push
84 
85   // Collect code coverage points
86   if (p.coverage) {
87     // count clock ticks to allow for coverage computation
88     val tick = true.B
89 
90     for (bit <- 0 to p.dataWidth - 1) {
91       cover(io.dataOut(bit)).suggestName(s"io_dataOut_$bit")
92       cover(io.dataIn(bit)).suggestName(s"io_dataIn_$bit")
93     }
94 
95     /* Ignore intentionally static inputs
96     for (bit <- 0 to log2Ceil(fifoDepth) - 1) {
97       cover(io.almostEmptyLevel(bit)).suggestName(s"io_almostEmptyLevel_$bit")
98       cover(io.almostFullLevel(bit)).suggestName(s"io_almostFullLevel_$bit")
99     }
100      */
101 
102     cover(tick).suggestName("tick")
103     cover(io.pop).suggestName("io__pop")
104     cover(io.push).suggestName("io__push")
105     cover(io.almostEmpty).suggestName("io__empty")
106     cover(io.almostFull).suggestName("io__full")
107     cover(io.almostEmpty).suggestName("io__almostEmpty")
108     cover(io.almostFull).suggestName("io__almostFull")
109 
110     if (p.externalRam) {
111       cover(io.ramWriteEnable).suggestName("io__ramWriteEnable")
112       cover(io.ramReadEnable).suggestName("io__ramReadEnable")
113       for (bit <- 0 to p.dataWidth - 1) {
114         cover(io.ramDataIn(bit)).suggestName(s"io_ramDataIn_$bit")
115         cover(io.ramDataOut(bit)).suggestName(s"io_ramDataOut_$bit")
116       }
117       for (bit <- 0 to log2Ceil(p.fifoDepth) - 1) {
118         cover(io.ramReadAddress(bit)).suggestName(s"io_ramReadAddress_$bit")
119         cover(io.ramWriteAddress(bit)).suggestName(s"io_ramWriteAddress_$bit")
120       }
121     }
122   }

125   /** Generate a basic SDC file for use with post-synthesis static timing
126     * analysis
127     *
128     * The illustration below shows the basic elements captured in an SDC file.
129     * 
130     * The follow variables are as follows:
131     *   period: inverse of the expected operating frequency 
132     *   dutyCycle: ratio of time when the clock is high and low (typ. 50%) 
133     *   inputDelay: time when the data is valid from the last rising clock edge 
134     *   outputDelay: time when the data is valid to the next rising clock edge
135     *
136     *   +-------+ 
137     * a-|       |
138     *   |  and  |--> c 
139     * b-|       | 
140     *   +-------+
141     *
142     *            |<--------->| Period 
143     *            |<--->|<--->| Duty Cycle 
144     *             _____       _____       _____ 
145     * clock _____|     |_____|     |_____|     |_____
146     *
147     *            |<->|       |<->| Input Delay 
148     *                 _______________________________
149     *     a _________|
150     *                             ___________________ 
151     *     b _____________________| 
152     * 
153     *                                |<->| Output Delay
154     *                                ________________ 
155     * c ____________________________|
156     *
157     */
158   //format:on 
159 
160   // Basic constraints
161   val period      = 5.000 // ns
162   val dutyCycle   = 0.50
163   val inputDelayPct = 0.2
164   val outputDelayPct = 0.2
165 
166   // Calculated constraints, override as needed below
167   val inputDelay  = period * inputDelayPct
168   val outputDelay = period * outputDelayPct
169   val fallingEdge = period * dutyCycle
170   val sdc         = new File("./syn/DynamicFifo.sdc")
171   val sdcFile     = new PrintWriter(sdc)
172 
173   val sdcFileData = s"""
174     create_clock -period $period -waveform {0 $fallingEdge} -name clock
175     set_input_delay -clock clock $inputDelay {reset}
176     set_input_delay -clock clock $inputDelay {io_pop}
177     set_input_delay -clock clock $inputDelay {io_push}
178     set_input_delay -clock clock $inputDelay {io_dataIn}
179     set_input_delay -clock clock $inputDelay {io_almostFullLevel}
180     set_input_delay -clock clock $inputDelay {io_almostEmptyLevel}
181     set_input_delay -clock clock $inputDelay {io_ramDataOut}
182     set_output_delay -clock clock $outputDelay {io_dataOut}
183     set_output_delay -clock clock $outputDelay {io_empty}
184     set_output_delay -clock clock $outputDelay {io_full}
185     set_output_delay -clock clock $outputDelay {io_almostEmpty}
186     set_output_delay -clock clock $outputDelay {io_almostFull}
187     set_output_delay -clock clock $outputDelay {io_ramDataIn}
188     set_output_delay -clock clock $outputDelay {io_ramWriteEnable}
189     set_output_delay -clock clock $outputDelay {io_ramReadEnable}
190   """.stripMargin
191 
192   sdcFile.write(s"${sdcFileData}")
193   sdcFile.close()
